//Обязательная строка для ВСЕХ ответов
OSSA-PTC: (код ответа)
//подключение к серверу
пришло:
	OSSA-PTC: 000f0010
	Hardware: (массив MAC-адресов того, кто подключается)
отправить:
	OSSA-PTC: (код ответа из connection stage)
//Авторизация
пришло:
	OSSA-PTC: 000f001c
	Username: (username)
	Pwd-hash: (хэш от пароля)
ответ:
	OSSA-PTC: 000f0a1c (авторизован) или 000f0a0c (не авторизован)
	Userid: (userid) (если авторизован)
//Добавить юзера
пришло:
	OSSA-PTC: 5c0f001c
	Username: (new user name)
	Pwd-clear: (password)
ответ:
	OSSA-PTC: 000f0b1c (всё ок) или faafff1c (ошибка)
	UID: (new user ID) (если всё ок)
	ERROR: (5 - не авторизован, 6 - нет доступа)
//Отправить сообщение
пришло:
	OSSA_PTC: 4c0f001c
	len: (длина сообщения)
	(JSON сообщения) {"body":"...","sender":"...","date":"..."}
отправить:
	OSSA-PTC: 0000f10a (ошибка) или 0000f11a (всё ок)
	ERROR: (5 - не авторизован, 6 - нет доступа)
	Total: всего сообщений
//Скачать все текущие сообщения
пришло:
	OSSA-PTC: 000bf11a
	From: (время в С-формате)
отправить:
	//все сообщения, дата отправлений которых БОЛЬШЕ времени из поля "From"
	OSSA-PTC: 000bf110 (нет сообщений или нет доступа) или 000bf11f (есть сообщения)
	Count: (count of messages) (если таковые есть)
	{JSON сообщений}
//Скачать список всех юзеров
пришло:
	OSSA-PTC: 000ba110
ответ:
	OSSA-PTC: 002ba110 (если авторизован)
	Count: (count of users)
	{JSON of user} {"username":"...","MAC":"..."} //поле MАС обязательно, но вместо реального МАС-а можно ставить ::::::::::
